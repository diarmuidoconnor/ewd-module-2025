### Exercise 4 (Iteration) solution

~~~tsx
type Module = { name: string; noLectures: number; noPracticals: number };

type CourseModulesProps = {
  course: string;
  modules: Module[];
};

export default function Demo({ course, modules }: CourseModulesProps) {
  const rows = modules.map((module, index) => (
    <tr key={index}>
      <td>{module.name}</td>
      <td>{module.noLectures}</td>
      <td>{module.noPracticals}</td>
    </tr>
  ));

  return (
    <div>
      <h1>{course}</h1>
      <table className="table table-bordered">
        <thead>
          <tr>
            <th>Name</th>
            <th>No lectures</th>
            <th>No practicals</th>
          </tr>
        </thead>
        <tbody>{rows}</tbody>
      </table>
    </div>
  );
}
~~~
